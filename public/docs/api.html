🔑 Authentication

Mỗi request cần header:

x-api-key: ugw_your_api_key_here

Lấy API Key từ admin → Tab "Đối tác" → Tạo đối tác mới.

6. Webhook Callback

Khi trạng thái thay đổi, hệ thống gửi POST đến callbackUrl của partner.

Headers:

Content-Type: application/json
x-signature: <HMAC-SHA256 of body using secretKey>
x-timestamp: <ISO timestamp>

Body:

{
  "event": "deposit.confirmed",
  "orderCode": "UDEP-A1B2C3D4-E5F6",
  "amountUsdt": 100,
  "amountVnd": 2554200,
  "txHash": "abc123...",
  "status": "CONFIRMED",
  "timestamp": "2026-04-16T02:00:00.000Z"
}

Events:

EventMô tả
deposit.confirmedNạp thành công
withdrawal.completedRút thành công
withdrawal.rejectedRút bị từ chối

Xác thực Signature (Node.js):

const crypto = require('crypto');

function verifySignature(body, signature, secretKey) {
  const expected = crypto
    .createHmac('sha256', secretKey)
    .update(JSON.stringify(body))
    .digest('hex');
  return expected === signature;
}

// Trong endpoint callback:
app.post('/callback', (req, res) => {
  const sig = req.headers['x-signature'];
  if (!verifySignature(req.body, sig, 'YOUR_SECRET_KEY')) {
    return res.status(403).send('Invalid signature');
  }
  // Xử lý event...
  res.status(200).send('OK');
});

Xác thực Signature (PHP):

function verifyCallback($body, $signature, $secretKey) {
    $expected = hash_hmac('sha256', json_encode($body), $secretKey);
    return hash_equals($expected, $signature);
}

Xác thực Signature (Python):

import hmac, hashlib, json

def verify_callback(body: dict, signature: str, secret: str) -> bool:
    expected = hmac.new(
        secret.encode(), json.dumps(body).encode(), hashlib.sha256
    ).hexdigest()
    return hmac.compare_digest(expected, signature)

8. Ví dụ tích hợp nhanh

Node.js

const API = 'http://103.161.225.245:3002/api/v1';
const KEY = 'ugw_your_api_key';
const headers = { 'Content-Type': 'application/json', 'x-api-key': KEY };

// Tạo đơn nạp
const dep = await fetch(`${API}/deposit`, {
  method: 'POST', headers,
  body: JSON.stringify({ amount: 100, orderId: 'MY-001' })
}).then(r => r.json());

console.log('Ví nạp:', dep.data.walletAddress);
console.log('QR:', dep.data.qrCodeUrl);

// Polling trạng thái
const check = setInterval(async () => {
  const s = await fetch(`${API}/status/${dep.data.orderCode}`, { headers })
    .then(r => r.json());
  if (s.data.status === 'CONFIRMED') {
    console.log('✅ Nạp thành công!');
    clearInterval(check);
  }
}, 10000);

// Tạo đơn rút
const wd = await fetch(`${API}/withdraw`, {
  method: 'POST', headers,
  body: JSON.stringify({ amount: 50, toAddress: 'TVhCRW...' })
}).then(r => r.json());

PHP

$apiUrl = 'http://103.161.225.245:3002/api/v1';
$apiKey = 'ugw_your_api_key';

$ch = curl_init("$apiUrl/deposit");
curl_setopt_array($ch, [
    CURLOPT_POST => true,
    CURLOPT_RETURNTRANSFER => true,
    CURLOPT_HTTPHEADER => [
        'Content-Type: application/json',
        "x-api-key: $apiKey",
    ],
    CURLOPT_POSTFIELDS => json_encode([
        'amount' => 100,
        'orderId' => 'PHP-001',
    ]),
]);
$result = json_decode(curl_exec($ch), true);
curl_close($ch);

echo "Ví: " . $result['data']['walletAddress'];

Python

import requests

API = 'http://103.161.225.245:3002/api/v1'
headers = {'Content-Type': 'application/json', 'x-api-key': 'ugw_your_api_key'}

r = requests.post(f'{API}/deposit', headers=headers,
                   json={'amount': 100, 'orderId': 'PY-001'})
data = r.json()['data']
print(f"Ví: {data['walletAddress']}")
print(f"Mã: {data['orderCode']}")
